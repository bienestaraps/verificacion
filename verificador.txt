    <script>
        // ==========================================
        // CONFIGURACIÓN PERMANENTE
        // Pega tu URL de Google Sheets (CSV) aquí abajo entre las comillas
        // ==========================================
        const URL_PERMANENTE = "https://docs.google.com/spreadsheets/d/e/2PACX-1vRk4EFHM2YcWewmQjBjJRLupHAUCysDp4SSWA-E5SakdXBKBRu8sQHE1uFaaAStX2m5Dv1bai9KFVId/pub?output=csv"; 
        // ==========================================

        const rutInput = document.getElementById('rutInput');
        const searchBtn = document.getElementById('searchBtn');
        const resultArea = document.getElementById('resultArea');
        const resultTitle = document.getElementById('resultTitle');
        const resultMsg = document.getElementById('resultMsg');
        const resultIcon = document.getElementById('resultIcon');
        const messageArea = document.getElementById('messageArea');
        const sheetUrlInput = document.getElementById('sheetUrl');
        const memberDetails = document.getElementById('memberDetails');

        // Cargar URL inicial
        if (URL_PERMANENTE) {
            sheetUrlInput.value = URL_PERMANENTE;
        }

        rutInput.addEventListener('input', (e) => {
            let value = e.target.value.replace(/[^0-9kK]/g, '');
            if (value.length > 1) {
                let dv = value.slice(-1);
                let cuerpo = value.slice(0, -1);
                e.target.value = cuerpo + '-' + dv;
            } else {
                e.target.value = value;
            }
        });

        function getRawDigits(val) {
            if (!val) return "";
            return val.toString().replace(/["' ]/g, '').replace(/[^0-9kK]/g, '').toUpperCase().trim();
        }

        async function verifyMember() {
            const rawRutToFind = getRawDigits(rutInput.value);
            const csvUrl = sheetUrlInput.value || URL_PERMANENTE;

            if (!rawRutToFind || rawRutToFind.length < 2) {
                showMessage("Por favor, ingresa un RUT válido", "text-amber-500");
                return;
            }

            if (!csvUrl) {
                showMessage("Error: No hay una base de datos conectada.", "text-red-500");
                return;
            }

            setLoading(true);

            try {
                const response = await fetch(csvUrl);
                if (!response.ok) throw new Error('No se pudo conectar');
                
                let csvText = await response.text();
                csvText = csvText.replace(/^\uFEFF/, '');
                
                const lines = csvText.split(/\r?\n/).filter(line => line.trim() !== "");
                const separator = lines[0].includes(';') ? ';' : ',';

                let foundMember = null;

                for (let i = 0; i < lines.length; i++) {
                    const columns = lines[i].split(separator);
                    const match = columns.some(col => getRawDigits(col) === rawRutToFind);

                    if (match) {
                        foundMember = {
                            nombre: (columns[1] || 'Sin Nombre').replace(/"/g, '').trim(),
                            estado: (columns[2] || 'ACTIVO').replace(/"/g, '').trim()
                        };
                        break;
                    }
                }
                displayResult(foundMember);
            } catch (error) {
                showMessage("Error al conectar con la base de datos.", "text-red-500");
            } finally {
                setLoading(false);
            }
        }

        function displayResult(member) {
            resultArea.classList.remove('hidden');
            memberDetails.classList.add('hidden');
            
            if (member) {
                resultArea.className = "status-card p-6 rounded-2xl border-2 flex flex-col items-center text-center space-y-4 bg-emerald-50 border-emerald-200 text-emerald-900 shadow-sm";
                resultTitle.innerText = "Socio/a Activo/a";
                resultMsg.innerText = member.nombre;
                resultIcon.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" class="h-14 w-14 text-emerald-600" fill="none" viewBox="0 0 24 24" stroke="currentColor"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="3" d="M5 13l4 4L19 7" /></svg>`;
                memberDetails.classList.remove('hidden');
                memberDetails.innerHTML = `<span class="text-emerald-600">${member.estado}</span>`;
            } else {
                resultArea.className = "status-card p-6 rounded-2xl border-2 flex flex-col items-center text-center space-y-4 bg-rose-50 border-rose-200 text-rose-900 shadow-sm";
                resultTitle.innerText = "No Registrado";
                resultMsg.innerText = "El RUT ingresado no figura en nuestra nómina actual.";
                resultIcon.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" class="h-14 w-14 text-rose-600" fill="none" viewBox="0 0 24 24" stroke="currentColor"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="3" d="M6 18L18 6M6 6l12 12" /></svg>`;
            }
        }

        function showMessage(msg, colorClass) {
            messageArea.innerText = msg;
            messageArea.className = `text-sm text-center py-2 font-bold ${colorClass}`;
            messageArea.classList.remove('hidden');
            setTimeout(() => messageArea.classList.add('hidden'), 4000);
        }

        function setLoading(isLoading) {
            if (isLoading) {
                searchBtn.disabled = true;
                searchBtn.classList.add('opacity-50');
                document.getElementById('btnText').innerText = "Buscando...";
                resultArea.classList.add('hidden');
            } else {
                searchBtn.disabled = false;
                searchBtn.classList.remove('opacity-50');
                document.getElementById('btnText').innerText = "Verificar";
            }
        }

        searchBtn.addEventListener('click', verifyMember);
        rutInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') verifyMember();
        });
    </script>
